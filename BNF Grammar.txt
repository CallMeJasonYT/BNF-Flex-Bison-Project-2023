file ::= comment RelativeLayout | comment LinearLayout

LinearLayout ::= LANGLE LINEAR
              reqattr1 id orientation
              RANGLE
              contentp
              LANGLE SLASH LINEAR RANGLE

RelativeLayout ::= LANGLE RELATIVE
                reqattr1 id
                RANGLE
                contents
                LANGLE SLASH RELATIVE RANGLE

TextView ::= LANGLE TEXTVIEW
          reqattr2 id textColor
          SLASH RANGLE

ImageView ::= LANGLE IMAGEVIEW
           reqattr3 id padding
           SLASH RANGLE
              
Button ::= LANGLE BUTTON
        reqattr2 id padding
        SLASH RANGLE

RadioGroup ::= LANGLE RADIOG
            reqattr1 id checkedButton
            RANGLE
            RadioButton
            LANGLE SLASH RADIOG RANGLE

RadioButton ::= RadioButton tempRB
             | tempRB
tempRB ::= comment | LANGLE BUTTON reqattr2 id SLASH RANGLE

ProgressBar ::= LANGE PROGRESSB 
             reqattr1 id max progress
             SLASH RANGLE
                   
reqattr1 ::= lwidth lheight
reqattr2 ::= lwidth lheight text
reqattr3 ::= lheight lwidth src

contentp ::= contentp content
          | content
contents ::= /*empty*/
          | contents content
          | content

content ::= LinearLayout | RelativeLayout | TextView | ImageView | Button | RadioGroup | RadioButton | ProgressBar | comment
elem ::= INTEGER | STRING

comment: /*empty*/
         | COMO comment_content COMC;

comment_content: /*empty*/
                 | comment_text comment_content 
                 | '-' comment_text comment_content;
comment_text: CHAR;

lwidth ::= LWIDTH elem DQUOTES
lheight ::= LHEIGHT elem DQUOTES
text ::= TEXT STRING DQUOTES
src ::= SRC STRING DQUOTES

id ::= /*empty*/
    | ID STRING DQUOTES
orientation ::= /*empty*/
             | ORIENTATION STRING DQUOTES
tcolor ::= /*empty*/
        | TEXTCOLOR STRING DQUOTES
padding ::= /*empty*/
        | PADDING INTEGER DQUOTES
cbutton ::= /*empty*/
         | CHECKEDBUTTON STRING DQUOTES
max ::= /*empty*/
     | MAX INTEGER DQUOTES
progress ::= /*empty*/
          | PROGRESS INTEGER DQUOTES
Char ::= %x00-7F

TOKENS:
-LWIDTH = "\ android:layout_width=\"" {ECHO return LWIDTH}
-LHEIGHT = "\ android:layout_height=\"" {ECHO return LHEIGHT}
-ID = "\ android:id=\"" {ECHO return ID}
-ORIENTATION = "\ android:orientation=\"" {ECHO return ORIENTATION}
-TEXT = "\ android:text=\"" {ECHO return TEXT}
-TEXTCOLOUR = "\ android:textColour=\"" {ECHO return TEXTCOLOUR}
-SRC = "\ android:src=\"" {ECHO return SRC}
-PADDING = "\ android:padding=\"" {ECHO return PADDING}
-CHECKEDB = "\ android:checkedButton=\"" {ECHO return CHECKEDB}
-MAX = "\ android:max=\"" {ECHO return MAX}
-PROGRESS = "\ android=progress=\"" {ECHO return PROGRESS}

-RELATIVE = RelativeLayout
-LINEAR = LinearLayout
-TEXTVIEW = TextView
-IMAGEVIEW = ImageView
-BUTTON = Button
-RADIOG = RadioGroup
-RADIOB = RadioButton
-PROGRESSB = ProgressBar
-INTEGER = int
-STRING = string
-SLASH = "/"
-LANGLE = "<"
-RANGLE = ">"
-DQUOTES = """
